-- Top 3 product categories by revenue in each region
-- Uses a CTE + window function to rank within each region

WITH sales_base AS (
    SELECT
        product_category,
        region,
        SUM(quantity_sold)                                       AS total_units,
        SUM(revenue)                                             AS total_revenue,
        ROUND(SUM(revenue) / NULLIF(SUM(quantity_sold), 0), 2)  AS avg_price_per_unit
    FROM   sales_transactions
    WHERE  transaction_date BETWEEN '2024-01-01' AND '2024-12-31'
      AND  status = 'Completed'
    GROUP  BY product_category, region
),
ranked AS (
    SELECT
        *,
        RANK() OVER (
            PARTITION BY region
            ORDER BY     total_revenue DESC
        ) AS revenue_rank
    FROM sales_base
)
SELECT *
FROM   ranked
WHERE  revenue_rank <= 3
ORDER  BY region, revenue_rank;
import pandas as pd
from sklearn.preprocessing import StandardScaler
from sklearn.cluster      import KMeans
import datetime

# Load transaction history
df = pd.read_csv('transactions.csv', parse_dates=['order_date'])
ref_date = df['order_date'].max() + datetime.timedelta(days=1)

# Build RFM table
rfm = df.groupby('customer_id').agg(
    recency   = ('order_date',  lambda x: (ref_date - x.max()).days),
    frequency = ('order_id',    'count'),
    monetary  = ('order_value', 'sum')
).reset_index()

# Scale features, then cluster into 4 segments
scaler = StandardScaler()
X      = scaler.fit_transform(rfm[['recency', 'frequency', 'monetary']])

km = KMeans(n_clusters=4, random_state=42, n_init=10)
rfm['segment'] = km.fit_predict(X)

# Map cluster IDs to human-readable labels
labels = {0: 'Champions', 1: 'At-Risk',
           2: 'Potential Loyalists', 3: 'Dormant'}
rfm['segment_name'] = rfm['segment'].map(labels)

# Export for stakeholders as a pivot-ready Excel file
rfm.to_excel('customer_segments.xlsx', index=False)
print(rfm.groupby('segment_name')['customer_id'].count())
